BÖLÜM 1: TEMEL KAVRAMLAR

1.1 React Native Nedir?

React Native, JavaScript ile iOS ve Android uygulamaları geliştirmeni sağlayan bir framework’tür. Tek kod yazarsın, hem iPhone hem Android’de çalışır.

// Normal React (Web için)
<div>Merhaba</div>

// React Native (Mobil için)
<View>
  <Text>Merhaba</Text>
</View>

1.2 TypeScript Nedir?

TypeScript, JavaScript’e “tip güvenliği” ekler. Hataları kod yazarken yakalar.

// JavaScript - Hata çalışma zamanında çıkar
let isim = "Mehrdad";
isim = 123; // Hata vermez ama sorun çıkarır

// TypeScript - Hata yazarken çıkar
let isim: string = "Mehrdad";
isim = 123; // ❌ HATA: Type 'number' is not assignable to type 'string'

1.3 Expo Nedir?

Expo, React Native geliştirmeyi kolaylaştıran bir platformdur.

  • Kamera, konum, bildirim gibi özellikleri hazır sunar
  • QR kod ile telefonda test edebilirsin
  • Build işlemlerini basitleştirir

BÖLÜM 5: VERİ DEPOLAMA (AsyncStorage)

5.1 AsyncStorage Nedir?

AsyncStorage, telefonun yerel hafızasına veri kaydetmeni sağlar. Uygulama kapansa bile veriler kalır. Web’deki localStorage gibi.

import AsyncStorage from "@react-native-async-storage/async-storage";

// ═══════════════════════════════════════════════════════════
// VERİ KAYDETME
// ═══════════════════════════════════════════════════════════

// String kaydetme
await AsyncStorage.setItem("kullanici_adi", "Mehrdad");

// Object/Array kaydetme (JSON'a çevirmek gerekir)
const favoriler = ["terim1", "terim2", "terim3"];
await AsyncStorage.setItem("favoriler", JSON.stringify(favoriler));

// ═══════════════════════════════════════════════════════════
// VERİ OKUMA
// ═══════════════════════════════════════════════════════════

// String okuma
const isim = await AsyncStorage.getItem("kullanici_adi");
// isim = 'Mehrdad'

// Object/Array okuma (JSON'dan çevirmek gerekir)
const favoriJSON = await AsyncStorage.getItem("favoriler");
const favoriler = favoriJSON ? JSON.parse(favoriJSON) : [];
// favoriler = ['terim1', 'terim2', 'terim3']

// ═══════════════════════════════════════════════════════════
// VERİ SİLME
// ═══════════════════════════════════════════════════════════

await AsyncStorage.removeItem("kullanici_adi");

BÖLÜM 6: CONTEXT API (Global State Yönetimi)

6.1 Context Nedir?

Context, verileri tüm component’lere “prop drilling” yapmadan ulaştırır.

❌ PROP DRILLING (Kötü yöntem):
App → HomeView → TermList → TermCard → term verisi

✅ CONTEXT (İyi yöntem):
App (Provider) ←→ Herhangi bir component (Consumer)

6.2 PharmacyContext.tsx - Satır Satır Açıklama

// src/context/PharmacyContext.tsx

// ═══════════════════════════════════════════════════════════
// IMPORT'LAR
// ═══════════════════════════════════════════════════════════

import React, {
  createContext, // Context oluşturmak için
  useContext, // Context'i kullanmak için
  useState, // State (değişken) tutmak için
  useEffect, // Yan etkiler (API çağrısı vb.) için
  useCallback, // Fonksiyonları optimize etmek için
} from "react";

import { pharmacyTermService } from "../services/PharmacyTermService";
// Terim işlemlerini yapan servis

// ═══════════════════════════════════════════════════════════
// CONTEXT TİPİ - Ne sunacağımızı tanımlıyoruz
// ═══════════════════════════════════════════════════════════

interface PharmacyContextType {
  // STATE (Veriler)
  terms: PharmacyTerm[]; // Tüm terimler
  isLoading: boolean; // Yükleniyor mu?
  error: string | null; // Hata mesajı
  searchText: string; // Arama metni
  searchResults: SearchResult[]; // Arama sonuçları

  // ACTIONS (Fonksiyonlar)
  setSearchText: (text: string) => void;
  searchTerms: (query: string) => Promise<void>;
  getTermsByCategory: (category: TermCategory) => Promise<PharmacyTerm[]>;
  toggleBookmark: (termId: string) => Promise<void>;
  getBookmarkedTerms: () => Promise<PharmacyTerm[]>;
  refresh: () => Promise<void>;
}

// ═══════════════════════════════════════════════════════════
// CONTEXT OLUŞTURMA
// ═══════════════════════════════════════════════════════════

const PharmacyContext = createContext<PharmacyContextType | undefined>(
  undefined
);
// undefined: Başlangıçta değer yok, Provider içinde set edilecek

// ═══════════════════════════════════════════════════════════
// PROVIDER COMPONENT - Veriyi sağlayan
// ═══════════════════════════════════════════════════════════

export const PharmacyProvider: React.FC<{ children: React.ReactNode }> = ({
  children,
}) => {
  // STATE TANIMLARI
  const [terms, setTerms] = useState<PharmacyTerm[]>([]);
  // terms: Tüm terimler, başlangıçta boş array

  const [isLoading, setIsLoading] = useState(false);
  // isLoading: Veri yüklenirken true olur

  const [error, setError] = useState<string | null>(null);
  // error: Hata varsa mesajı tutar

  const [searchText, setSearchText] = useState("");
  // searchText: Kullanıcının yazdığı arama metni

  const [searchResults, setSearchResults] = useState<SearchResult[]>([]);
  // searchResults: Arama sonuçları

  // ═══════════════════════════════════════════════════════════
  // useEffect - Component mount olduğunda çalışır
  // ═══════════════════════════════════════════════════════════

  useEffect(() => {
    loadAllTerms(); // Uygulama açıldığında tüm terimleri yükle
  }, []); // [] = Sadece bir kez çalış

  // ═══════════════════════════════════════════════════════════
  // TERİMLERİ YÜKLEME FONKSİYONU
  // ═══════════════════════════════════════════════════════════

  const loadAllTerms = async () => {
    setIsLoading(true); // Yükleniyor...
    setError(null); // Önceki hatayı temizle

    try {
      // Servisten tüm terimleri al
      const allTerms = await pharmacyTermService.getAllTerms();

      // Verileri normalize et (tutarlı hale getir)
      const normalizedTerms = allTerms.map((term) => ({
        ...term,
        isBookmarked: Boolean(term.isBookmarked), // true/false garantile
        createdAt:
          term.createdAt instanceof Date
            ? term.createdAt
            : new Date(term.createdAt), // Date objesi garantile
        components: Array.isArray(term.components) ? term.components : [], // Array garantile
      }));

      setTerms(normalizedTerms); // State'e kaydet
    } catch (err) {
      // Hata yakalama
      setError(err instanceof Error ? err.message : "Hata oluştu");
    } finally {
      setIsLoading(false); // Yükleme bitti
    }
  };

  // ═══════════════════════════════════════════════════════════
  // FAVORİ TOGGLE FONKSİYONU
  // ═══════════════════════════════════════════════════════════

  const toggleBookmark = useCallback(async (termId: string) => {
    try {
      // Serviste favori durumunu değiştir
      const newBookmarkState = await pharmacyTermService.toggleBookmark(termId);

      // Local state'i güncelle
      setTerms((prevTerms) =>
        prevTerms.map((term) =>
          term.id === termId
            ? { ...term, isBookmarked: Boolean(newBookmarkState) }
            : term
        )
      );
    } catch (err) {
      setError("Favori eklenirken hata oluştu");
    }
  }, []);
  // useCallback: Fonksiyonu optimize eder, gereksiz yeniden oluşturmayı önler

  // ═══════════════════════════════════════════════════════════
  // CONTEXT DEĞERİ
  // ═══════════════════════════════════════════════════════════

  const value: PharmacyContextType = {
    terms,
    isLoading,
    error,
    searchText,
    searchResults,
    setSearchText,
    searchTerms: performSearch,
    getTermsByCategory,
    toggleBookmark,
    getBookmarkedTerms,
    refresh: loadAllTerms,
  };

  // Provider ile children'ı sar
  return (
    <PharmacyContext.Provider value={value}>
      {children}
    </PharmacyContext.Provider>
  );
};

// ═══════════════════════════════════════════════════════════
// CUSTOM HOOK - Context'i kullanmak için
// ═══════════════════════════════════════════════════════════

export const usePharmacy = () => {
  const context = useContext(PharmacyContext);

  // Provider dışında kullanılırsa hata ver
  if (context === undefined) {
    throw new Error("usePharmacy must be used within a PharmacyProvider");
  }

  return context;
};

export const ThemeProvider: React.FC<{ children: React.ReactNode }> = ({
  children,
}) => {
  // Tema state'i - varsayılan: dark
  const [theme, setThemeState] = useState<"light" | "dark">("dark");

  // Uygulama açıldığında kayıtlı temayı yükle
  useEffect(() => {
    loadTheme();
  }, []);

  const loadTheme = async () => {
    try {
      // AsyncStorage'dan temayı oku
      const savedTheme = await AsyncStorage.getItem("pharmadict_theme");
      if (savedTheme === "light" || savedTheme === "dark") {
        setThemeState(savedTheme);
      }
    } catch (error) {
      console.error("Tema yüklenirken hata:", error);
    }
  };

  // Tema değiştirme fonksiyonu
  const toggleTheme = useCallback(async () => {
    const newTheme = theme === "light" ? "dark" : "light";
    setThemeState(newTheme);
    // AsyncStorage'a kaydet
    await AsyncStorage.setItem("pharmadict_theme", newTheme);
  }, [theme]);

  // Aktif tema renklerini seç
  const colors = theme === "light" ? lightTheme : darkTheme;
  const isDark = theme === "dark";

  return (
    <ThemeContext.Provider value={{ theme, colors, toggleTheme, isDark }}>
      {children}
    </ThemeContext.Provider>
  );
};

export const useTheme = () => {
  const context = useContext(ThemeContext);
  if (context === undefined) {
    throw new Error("useTheme must be used within a ThemeProvider");
  }
  return context;
};

  // ═══════════════════════════════════════════════════════════
  // BAŞLATMA - AsyncStorage'dan notları yükle
  // ═══════════════════════════════════════════════════════════

  async initialize(): Promise<void> {
    if (this.initialized) return;

    try {
      const stored = await AsyncStorage.getItem("pharmadict_notes");

      if (stored) {
        const parsed = JSON.parse(stored);

        // Object'i Map'e dönüştür
        Object.entries(parsed).forEach(([termId, note]: [string, any]) => {
          this.notes.set(termId, {
            ...note,
            createdAt: new Date(note.createdAt), // String'den Date'e
            updatedAt: new Date(note.updatedAt),
          });
        });
      }

      this.initialized = true;
    } catch (error) {
      console.error("Notlar yüklenirken hata:", error);
    }
  }

  // ═══════════════════════════════════════════════════════════
  // STORAGE'A KAYDET
  // ═══════════════════════════════════════════════════════════

  private async saveToStorage(): Promise<void> {
    // Map'i Object'e dönüştür (JSON için)
    const obj: { [key: string]: TermNote } = {};
    this.notes.forEach((note, termId) => {
      obj[termId] = note;
    });

    await AsyncStorage.setItem("pharmadict_notes", JSON.stringify(obj));
  }

  // ═══════════════════════════════════════════════════════════
  // NOT GETİR
  // ═══════════════════════════════════════════════════════════

  }

  // ═══════════════════════════════════════════════════════════
  // NOT KAYDET
  // ═══════════════════════════════════════════════════════════

  async saveNote(termId: string, content: string): Promise<TermNote> {
    await this.initialize();

    const existingNote = this.notes.get(termId);
    const now = new Date();

    const note: TermNote = {
      termId,
      content,
      createdAt: existingNote?.createdAt || now, // Varsa eskisini koru
      updatedAt: now,
    };

    if (content.trim() === "") {
      // Boş not ise sil
      this.notes.delete(termId);
    } else {
      this.notes.set(termId, note);
    }

    await this.saveToStorage();
    return note;
  }

  // ═══════════════════════════════════════════════════════════
  // NOT SİL
  // ═══════════════════════════════════════════════════════════

  async deleteNote(termId: string): Promise<void> {
    await this.initialize();
    this.notes.delete(termId);
    await this.saveToStorage();
  }

  // ═══════════════════════════════════════════════════════════
  // NOTLU TERİM ID'LERİ (Favorilerde ikon göstermek için)
  // ═══════════════════════════════════════════════════════════

    await this.initialize();
    return Array.from(this.notes.keys());
  }
}
